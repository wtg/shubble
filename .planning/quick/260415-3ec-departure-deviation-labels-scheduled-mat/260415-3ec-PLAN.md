```typescript
function getDepartureLabel(
  trip: Trip,
  routeTimes: string[],
  _selectedDay: number,
  timeToDate: (t: string) => Date,
): DepartureLabel | null {
  const TIME_FORMAT: Intl.DateTimeFormatOptions = { hour: 'numeric', minute: '2-digit' };

  // Format a Date as "5:11 PM" for inline label text. Strip the trailing
  // " AM"/" PM" when it matches the row anchor's meridiem to keep labels
  // short — but keeping it is also fine; simpler to always include.
  const fmt = (d: Date) => d.toLocaleTimeString(undefined, TIME_FORMAT);

  // --- Case 1: scheduled-matched trip ---
  if (trip.scheduled) {
    // actual_departure CAN be null (e.g. scheduled row with no vehicle yet).
    // In that case there's no deviation to compute — return null.
    if (!trip.actual_departure) return null;

    const scheduledMs = new Date(trip.departure_time).getTime();
    const actualMs = new Date(trip.actual_departure).getTime();
    const deltaMin = Math.round((actualMs - scheduledMs) / 60_000);

    // Dead zone: |delta| <= 1 min → no label
    if (Math.abs(deltaMin) <= 1) return null;

    const actualStr = fmt(new Date(actualMs));
    if (deltaMin > 0) {
      return {
        text: `(${actualStr}, +${deltaMin} min late)`,
        kind: 'matched-late',
      };
    } else {
      return {
        text: `(${actualStr}, ${Math.abs(deltaMin)} min early)`,
        kind: 'matched-early',
      };
    }
  }

  // --- Case 2: off-schedule (injected) trip ---
  // For injected trips, trip.departure_time === trip.actual_departure per backend
  // contract (trips.py). Use departure_time as the canonical "actual" to keep
  // this helper functional even if actual_departure is null for any reason.
  const actualMs = trip.actual_departure
    ? new Date(trip.actual_departure).getTime()
    : new Date(trip.departure_time).getTime();
  const actualDate = new Date(actualMs);

  if (routeTimes.length === 0) {
    return { text: 'Unscheduled', kind: 'unscheduled-far' };
  }

  // Find nearest scheduled slot by absolute ms-distance.
  let nearestSlotStr = routeTimes[0];
  let nearestDeltaMs = Math.abs(actualMs - timeToDate(routeTimes[0]).getTime());
  let nearestSignedDeltaMs = actualMs - timeToDate(routeTimes[0]).getTime();
  for (let i = 1; i < routeTimes.length; i++) {
    const slotMs = timeToDate(routeTimes[i]).getTime();
    const absD = Math.abs(actualMs - slotMs);
    if (absD < nearestDeltaMs) {
      nearestDeltaMs = absD;
      nearestSignedDeltaMs = actualMs - slotMs;
      nearestSlotStr = routeTimes[i];
    }
  }

  const nearestMin = Math.round(nearestDeltaMs / 60_000);

  // >30 min from any slot → plain "Unscheduled"
  if (nearestMin > 30) {
    return { text: 'Unscheduled', kind: 'unscheduled-far' };
  }

  // Within 30 min: arrow pill anchored to nearest slot.
  // actual BEFORE nearest slot (signed < 0) → ↑ early to that slot
  // actual AFTER nearest slot (signed > 0)  → ↓ late from that slot
  // Exactly equal (0) → match fell on a slot; treat as "unscheduled" on-slot
  // (this is a degenerate case — shouldn't happen since the row would have
  //  matched instead of being injected, but guard anyway).
  if (nearestSignedDeltaMs === 0) {
    return { text: 'Unscheduled', kind: 'unscheduled-far' };
  }

  // Note: slot labels in routeTimes already include meridiem ("5:30 PM").
  // Re-format via timeToDate→fmt to normalize spacing ("5:30 PM" vs
  // "5:30PM") and guarantee consistency with actualStr formatting.
  const slotStr = fmt(timeToDate(nearestSlotStr));

  if (nearestSignedDeltaMs < 0) {
    return {
      text: `↑ ${nearestMin} min early to ${slotStr} slot`,
      kind: 'unscheduled-early',
    };
  }
  return {
    text: `↓ ${nearestMin} min late from ${slotStr} slot`,
    kind: 'unscheduled-late',
  };
}
```
